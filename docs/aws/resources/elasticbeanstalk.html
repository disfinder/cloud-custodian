terminate¶

Terminate an ElasticBeanstalk Environment.

Example

policies:
  - name: eb-env-termination
    resource: elasticbeanstalk-environment
    filters:
      - type: marked-for-op
        op: terminate
    actions:
      - terminate

properties:
  force:
    default: false
    type: boolean
  terminate_resources:
    default: true
    type: boolean
  type:
    enum:
    - terminate
required:
- type
